    def protocol
      if (@atr[0] & 0x80) == 0x00
        'ISO'
      elsif (@atr[0] & 0xf0) == 0x80
        'I2C'
      elsif (@atr[0] & 0xf0) == 0x90
        '3W'
      elsif (@atr[0] & 0xf0) == 0xa0
        '2W'
      else
        'unknown'
      end
    end

    def structure
      if (@atr[0] & 0x3) == 0
        'ISO'
      elsif (@atr[0] & 0x7) == 2
        'common use'
      elsif (@atr[0] & 0x7) == 6
        'proprietary use'
      else
        'special use'
      end
    end
